def os
  host_os = RbConfig::CONFIG["host_os"]
  case host_os
  when /mswin|msys|mingw|cygwin|bccwin|wince|emc/
    "Windows #{host_os}"
  when /darwin|mac os/
    "MacOS #{host_os}"
  when /linux/
    "Linux #{host_os}"
  when /solaris|bsd/
    "Unix #{host_os}"
  else
    "Unknown #{host_os}"
  end
end
